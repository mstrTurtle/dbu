processBranch cwd:
  if branch is dev or master:
    cwd += branch
    return 0
  cwd += branch
  fs = findFolders cwd subbranch
  case fs:
	[] -> return ERR_NO_ENTRY
	[f] -> cwd += f
	other -> return ERR_CANNOT_KNOW
